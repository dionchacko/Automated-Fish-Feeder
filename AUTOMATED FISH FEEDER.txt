/* ---------- FEED FUNCTION ---------- */

void feedFish()
{
  DateTime now = rtc.now();

  String event =
  String(now.day()) + "/" +
  String(now.month()) + " " +
  String(now.hour()) + ":" +
  String(now.minute()) + ":" +
  String(now.second()) +
  " - Feeding Done";

  pastEvents += event + "<br>";

  lcd.clear();
  lcd.setCursor(0,0);
  lcd.print("FEEDING...");

  feederServo.write(OPEN_POS);
  delay(350);

  feederServo.write(CLOSE_POS);

  delay(1500);
  lcd.clear();
}

/* ---------- WEB ROUTES ---------- */

void handleRoot()
{
  server.send_P(200,"text/html",webpage);
}

void handleFeed()
{
  feedFish();
  server.send(200,"text/plain","Feeding Done");
}

void handleSchedule()
{
  sch_day = server.arg("day").toInt();
  sch_month = server.arg("month").toInt();
  sch_hour = server.arg("hour").toInt();
  sch_minute = server.arg("minute").toInt();

  scheduleActive=true;

  nextEvent =
  String(sch_day)+"/"+
  String(sch_month)+" "+
  String(sch_hour)+":"+
  String(sch_minute);

  server.send(200,"text/plain","Schedule Saved");
}

/* ---------- SETUP ---------- */

void setup()
{

Serial.begin(115200);

Wire.begin();
rtc.begin();

lcd.init();
lcd.backlight();

feederServo.attach(servoPin);
feederServo.write(CLOSE_POS);

/* WIFI */

Serial.println("Connecting to WiFi");

WiFi.begin(ssid,password);

while(WiFi.status()!=WL_CONNECTED)
{
delay(500);
Serial.print(".");
}

Serial.println("");
Serial.println("WiFi Connected");

Serial.print("IP Address: ");
Serial.println(WiFi.localIP());

lcd.setCursor(0,0);
lcd.print("WiFi Connected");

delay(2000);
lcd.clear();

/* WEB SERVER */

server.on("/",handleRoot);
server.on("/feed",handleFeed);
server.on("/set",handleSchedule);

server.begin();

Serial.println("Server Started");

}

/* ---------- LOOP ---------- */

void loop()
{

server.handleClient();

DateTime now = rtc.now();

/* LCD CLOCK */

lcd.setCursor(0,0);

if(now.hour()<10) lcd.print("0");
lcd.print(now.hour());
lcd.print(":");

if(now.minute()<10) lcd.print("0");
lcd.print(now.minute());
lcd.print(":");

if(now.second()<10) lcd.print("0");
lcd.print(now.second());

lcd.setCursor(0,1);

if(now.day()<10) lcd.print("0");
lcd.print(now.day());
lcd.print("/");

if(now.month()<10) lcd.print("0");
lcd.print(now.month());
lcd.print("/");

lcd.print(now.year());

/* AUTO FEED */

if(millis()-lastFeed >= feedInterval)
{
feedFish();
lastFeed=millis();
}

/* SCHEDULE FEED */

if(scheduleActive)
{
if(now.day()==sch_day &&
   now.month()==sch_month &&
   now.hour()==sch_hour &&
   now.minute()==sch_minute &&
   now.second()==0)
{
feedFish();
}
}
